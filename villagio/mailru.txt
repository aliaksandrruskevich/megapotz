var arr = [];

jQuery('.b-datalist__item').each(function() {
	if( jQuery(this).find('.b-datalist__item__addr').text() == 'portal@rosreestr.ru') {
		arr.push(
			jQuery(this).find('.b-datalist__item__subj').text().replace(/.+(\d\d\-\d+).+/, '$1')
			+ "	" +
			jQuery(this).find('.b-datalist__item__date__value').attr('title')
		);
	}
});

console.info(arr.join("\n"));
